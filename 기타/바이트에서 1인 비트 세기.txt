#include <iostream>
#include <algorithm>
#include<string>
#include <ctime>
#include <cstdlib>
#include<cstring>
using namespace std;


int count_byte(unsigned char cpy)
{
	unsigned char cnt = 0;
	
	while (cpy)
	{
		cnt += cpy & (unsigned char)1;
		cpy = cpy >> 1;
	}

	return cnt;
}

int main(void)
{
	while (true)
	{
		unsigned a;
		cout << "정수 입력 :";
		cin >> a;
		cout << count_byte(a) << endl;
	}

	return 0;
}
