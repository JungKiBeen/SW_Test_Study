#include <stdio.h>

int a[10000 + 10];
int N;

int prev_permt()
{
	int i, j, start, end, temp;
	i = N - 1;
	while ((i != 0) && a[i - 1] < a[i])		i--;

	if (i == 0) return 0;

	j = N - 1;
	while (a[j] > a[i - 1])		j--;

	temp = a[j];
	a[j] = a[i - 1];
	a[i - 1] = temp;

	start = i;	end = N - 1;

	while (start < end)
	{
		temp = a[start];
		a[start] = a[end];
		a[end] = temp;
		start++;	end--;
	}

	return 1;
}
int main()
{
	int i, j;
	freopen("input.txt", "r", stdin);

	scanf("%d", &N);
	for (i = 0; i < N; i++)
		scanf("%d", &a[i]);

	if (!prev_permt())
	{
		printf("%d", -1);
		return 0;
	}

	for (i = 0; i < N; i++)
	{
		printf("%d ", a[i]);
	}
	return 0;
}
